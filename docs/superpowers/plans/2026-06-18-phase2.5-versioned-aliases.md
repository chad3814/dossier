# Phase 2.5 — Versioned Aliases — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development or superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Version entity aliases by reading position (anchored, sparse) so a gated view no longer leaks alias spoilers — captured in the same re-describe pass that produces versioned descriptions.

**Architecture:** An `AliasEvent` stream (`dcc/log/aliases.json`); a leak-safe `materialize` overlay (processed-keyed) for both descriptions and aliases; the chapter agent additionally matches each entity's not-yet-seen known aliases against the chapter text; a deterministic reconciliation places unmatched aliases at the entity's last appearance.

**Tech Stack:** TypeScript (ESM, `.js` specifiers), `tsx`, `vitest`.

**Spec:** `docs/superpowers/specs/2026-06-18-phase2.5-versioned-aliases-design.md`
**Working dir:** `dossier/` (DCC data in `../dcc/`).

  // PROCESSED (has ≥1 event): processed → only events ≤ cutoff (never the blob, so an entity
  // that appears before its first version shows nothing rather than leaking the full blob);
  // unprocessed → keep the appearance-log blob.
  const byId = new Map(entities.map((e) => [e.id, e]));

  if (options.descriptions && options.descriptions.length > 0) {
    const latest = new Map<string, DescriptionEvent>();
    const processed = new Set<string>();
    for (const ev of options.descriptions) {
      if (!byId.has(ev.id)) continue;
      processed.add(ev.id);
      if (!withinCutoff(ev.anchor, upTo)) continue;
      const cur = latest.get(ev.id);
      if (!cur || cmpAnchor(normalizeAnchor(ev.anchor), normalizeAnchor(cur.anchor)) > 0) latest.set(ev.id, ev);
    }
    for (const id of processed) {
      const e = byId.get(id) as RegistryEntity;
      const ev = latest.get(id);
      e.description = ev ? ev.description : "";
      if (ev) e.significance = ev.significance;
    }
  }

  if (options.aliases && options.aliases.length > 0) {
    const collected = new Map<string, string[]>();
    const processed = new Set<string>();
    for (const ev of options.aliases) {
      if (!byId.has(ev.id)) continue;
      processed.add(ev.id);
      if (!withinCutoff(ev.anchor, upTo)) continue;
      const list = collected.get(ev.id) ?? [];
      list.push(ev.alias);
      collected.set(ev.id, list);
    }
    for (const id of processed) {
      const e = byId.get(id) as RegistryEntity;
      e.aliases = cleanAliases(e.canonicalName, collected.get(id) ?? []);
    }
  }

phase('Redescribe');
const state = new Map(Object.entries(seed));
const emitted = new Map();
const descriptionEvents = [];
const aliasEvents = [];
for (let i = 0; i < chapters.length; i++) {
  const ch = chapters[i];
  if (!ch.entities.length) continue;
  const withCurrent = ch.entities.map((e) => ({
    id: e.id, canonicalName: e.canonicalName, type: e.type,
    current: state.get(e.id) ? state.get(e.id).description : null,
    candidateAliases: (aliasesById[e.id] || []).filter((a) => !(emitted.get(e.id) && emitted.get(e.id).has(a))),
  }));
  const res = await agentRetry(prompt(`${sectionDir}/${ch.file}`, withCurrent), { schema, phase: 'Redescribe', label: `redescribe:B${bookNumber}·${ch.label}` });
  if (!res) continue;
  const anchorById = new Map(ch.entities.map((e) => [e.id, e.anchor]));
  for (const u of (res.descriptions || [])) {
    const anchor = anchorById.get(u.id);
    if (!anchor) continue;
    descriptionEvents.push({ id: u.id, anchor, description: u.description, significance: u.significance });
    state.set(u.id, { description: u.description, significance: u.significance });
  }
  for (const a of (res.aliases || [])) {
    const anchor = anchorById.get(a.id);
    if (!anchor) continue;
    const cand = new Set(aliasesById[a.id] || []);
    let set = emitted.get(a.id); if (!set) { set = new Set(); emitted.set(a.id, set); }
    for (const alias of (a.seen || [])) {
      if (!cand.has(alias) || set.has(alias)) continue;
      aliasEvents.push({ id: a.id, anchor, alias });
      set.add(alias);
    }
  }
  if ((i + 1) % 10 === 0 || i === chapters.length - 1) log(`B${bookNumber}: ${i + 1}/${chapters.length} chapters, ${descriptionEvents.length} desc, ${aliasEvents.length} alias versions`);
}
log(`Done: book ${bookNumber}, ${descriptionEvents.length} description + ${aliasEvents.length} alias versions`);
return { bookNumber, descriptionEvents, aliasEvents };

```ts
import { anchorSortKey, normalizeAnchor, normalizeName } from "./registry.js";
import type { AliasEvent, Registry } from "./types.js";

function cmp(a: string, b: string): number {
  const ka = anchorSortKey(a), kb = anchorSortKey(b);
  return ka[0] - kb[0] || ka[1] - kb[1] || ka[2] - kb[2];
}

/**
 * Ensure every registry alias is represented: any alias the re-describe pass never matched in
 * text is added at the entity's LAST appearance anchor (present at full-series, never early).
 * Returns the merged event list sorted by anchor then id.
 */
export function reconcileAliases(registry: Registry, events: AliasEvent[]): AliasEvent[] {
  const emittedByn = new Map<string, Set<string>>();
  for (const e of events) {
    const set = emittedByn.get(e.id) ?? new Set<string>();
    set.add(normalizeName(e.alias));
    emittedByn.set(e.id, set);
  }
  const out = [...events];
  for (const ent of registry.entities) {
    const seen = emittedByn.get(ent.id) ?? new Set<string>();
    const last = [...ent.appearances].map(normalizeAnchor).sort(cmp).pop();
    if (!last) continue;
    for (const alias of ent.aliases) {
      if (seen.has(normalizeName(alias))) continue;
      out.push({ id: ent.id, anchor: last, alias });
      seen.add(normalizeName(alias));
    }
  }
  out.sort((a, b) => cmp(normalizeAnchor(a.anchor), normalizeAnchor(b.anchor)) || a.id.localeCompare(b.id));
  return out;
}
```

```ts
export function validateAliases(registry: Registry, events: AliasEvent[]): { errors: string[]; warnings: string[] } {
  const errors: string[] = [];
  const byId = new Map(registry.entities.map((e) => [e.id, e]));
  const grouped = new Map<string, AliasEvent[]>();
  for (const ev of events) {
    if (!byId.has(ev.id)) { errors.push(`unknown entity id "${ev.id}" in alias event @ ${ev.anchor}`); continue; }
    const list = grouped.get(ev.id) ?? []; list.push(ev); grouped.set(ev.id, list);
  }
  for (const [id, evs] of grouped) {
    for (let i = 1; i < evs.length; i++) {
      if (cmp(normalizeAnchor(evs[i]!.anchor), normalizeAnchor(evs[i - 1]!.anchor)) < 0) errors.push(`"${id}": alias events out of anchor order`);
    }
  }
  return { errors, warnings: [] };
}
```
